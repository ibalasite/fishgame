Usage Guide

OpenAPI / Swagger UI

# Validate
npx @redocly/openapi-cli lint docs/blueprint/contracts/openapi.yaml

# Generate client SDK (TypeScript — Cocos Creator 用)
npx openapi-generator-cli generate \
  -i docs/blueprint/contracts/openapi.yaml \
  -g typescript-axios \
  -o src/generated/client

# Start Swagger UI
npx swagger-ui-serve docs/blueprint/contracts/openapi.yaml

JSON Schema Validation

# Validate a payload against RegisterRequest schema
npx ajv validate \
  -s docs/blueprint/contracts/schemas/RegisterRequest.schema.json \
  -d payload.json

# Validate all schemas
for f in docs/blueprint/contracts/schemas/*.schema.json; do
  npx ajv compile -s "$f" && echo "✅ $f"
done

Deploy with Docker Compose

# Start all services
docker-compose up -d

# Run Prisma migrations
docker-compose exec api npx prisma migrate deploy

# Seed initial game_configs
docker-compose exec api npx prisma db seed

# View logs
docker-compose logs -f game
